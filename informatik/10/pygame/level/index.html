TL;DR

Sprite-Klassen bleiben eigenständig. Die Level-Klasse liest eine Textdatei ein und baut daraus alle Objekte. Exits/Backs sind Sprite-Gruppen (Multi-Tile). Level-Wechsel per Modulo.

level_dateien = ["level_1.txt", "level_2.txt", "level_3.txt"]
level_nr = 0
level = Level(level_dateien[level_nr])

# Vorwärts:  level_nr = (level_nr + 1) % len(level_dateien)
# Zurück:    level_nr = (level_nr - 1) % len(level_dateien)
level = Level(level_dateien[level_nr])

Idee

Deine Sprite-Klassen (Wall, Coin, …) bleiben eigenständig. Die Level-Klasse liest eine Textdatei ein und erzeugt daraus die Objekte. Das trennt Level-Design (Textdatei) sauber vom Code.

class Back(pygame.sprite.Sprite):

Da exits und backs Sprite-Gruppen sind, funktioniert die Kollision auch wenn der Ausgang aus 2×2 oder 3×1 Tiles besteht – spritecollideany prüft gegen jedes Sprite in der Gruppe.

Die Level-Klasse

class Level:
    def __init__(self, dateiname):
        self.walls   = pygame.sprite.Group()
        self.coins   = pygame.sprite.Group()
        self.enemies = pygame.sprite.Group()
        self.exits   = pygame.sprite.Group()
        self.backs   = pygame.sprite.Group()
        self.player  = None
        self._laden(dateiname)

    def _laden(self, dateiname):
        with open(dateiname) as datei:
            for zeile_nr, zeile in enumerate(datei):
                for spalte_nr, zeichen in enumerate(zeile.strip()):
                    x = spalte_nr * TILE_SIZE
                    y = zeile_nr  * TILE_SIZE

                    if zeichen == "W":
                        self.walls.add(Wall(x, y))
                    elif zeichen == "C":
                        self.coins.add(Coin(x, y))
                    elif zeichen == "E":
                        self.enemies.add(Enemy(x, y))
                    elif zeichen == "S":
                        self.player = Player(x, y)
                    elif zeichen == "X":
                        self.exits.add(Exit(x, y))
                    elif zeichen == "B":
                        self.backs.add(Back(x, y))

    def update(self):
        self.player.update(self.walls)
        self.enemies.update(self.walls)
        gesammelt = pygame.sprite.spritecollide(
            self.player, self.coins, True
        )
        return len(gesammelt)

    def draw(self, screen):
        self.walls.draw(screen)
        self.coins.draw(screen)
        self.enemies.draw(screen)
        self.exits.draw(screen)
        self.backs.draw(screen)
        screen.blit(self.player.image, self.player.rect)

    def ausgang_erreicht(self):
        return pygame.sprite.spritecollideany(
            self.player, self.exits
        )

    def rueckgang_erreicht(self):
        return pygame.sprite.spritecollideany(
            self.player, self.backs
        )

Level wechseln mit Modulo

level_dateien = ["level_1.txt", "level_2.txt", "level_3.txt"]
level_nr = 0
level = Level(level_dateien[level_nr])

# Im Game Loop:
if level.ausgang_erreicht():
    level_nr = (level_nr + 1) % len(level_dateien)
    level = Level(level_dateien[level_nr])

elif level.rueckgang_erreicht():
    level_nr = (level_nr - 1) % len(level_dateien)
    level = Level(level_dateien[level_nr])
Warum Modulo? (0 - 1) % 3 ergibt 2 – geht vom ersten Level zum letzten. (2 + 1) % 3 ergibt 0 – geht vom letzten zum ersten. Kein if nötig.

TL;DR

Alle Sprite-Klassen leben innerhalb von Level (nested classes). Die Objekte werden direkt mit Koordinaten platziert – keine Textdatei, kein Zeichenraster.

level = Level(0)       # Level per Nummer
level.update()
level.draw(screen)

# Intern: Level.Wall(x,y), Level.Coin(x,y), ...
# Positionen werden direkt in _aufbauen() gesetzt

Idee

Die Sprite-Klassen werden innerhalb der Level-Klasse definiert (nested classes). Statt einer Textdatei platzierst du Objekte direkt mit Pixelkoordinaten oder per Schleife. Alles steckt in einer Methode pro Level.

Wann sinnvoll? Wenn du volle Kontrolle über Positionen brauchst (z.B. Objekte nicht im Raster) und die Sprite-Klassen nur für das Level existieren.

Die Level-Klasse

class Level:

    # ---- Verschachtelte Sprite-Klassen ----

    class Wall(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface((TILE_SIZE, TILE_SIZE))
            self.image.fill("gray")
            self.rect = self.image.get_rect(topleft=(x, y))

    class Coin(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface(
                (TILE_SIZE, TILE_SIZE), pygame.SRCALPHA
            )
            pygame.draw.circle(
                self.image, "gold",
                (TILE_SIZE // 2, TILE_SIZE // 2), TILE_SIZE // 3
            )
            self.rect = self.image.get_rect(topleft=(x, y))

    class Enemy(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface((TILE_SIZE, TILE_SIZE))
            self.image.fill("red")
            self.rect = self.image.get_rect(topleft=(x, y))
            self.speed = 2
            self.direction = 1

        def update(self, walls):
            self.rect.x += self.speed * self.direction
            if pygame.sprite.spritecollideany(self, walls):
                self.rect.x -= self.speed * self.direction
                self.direction *= -1

    class Player(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface((TILE_SIZE, TILE_SIZE))
            self.image.fill("limegreen")
            self.rect = self.image.get_rect(topleft=(x, y))
            self.speed = 4

        def update(self, walls):
            keys = pygame.key.get_pressed()
            dx, dy = 0, 0
            if keys[pygame.K_LEFT]:  dx = -self.speed
            if keys[pygame.K_RIGHT]: dx =  self.speed
            if keys[pygame.K_UP]:    dy = -self.speed
            if keys[pygame.K_DOWN]:  dy =  self.speed

            self.rect.x += dx
            for w in pygame.sprite.spritecollide(self, walls, False):
                if dx > 0: self.rect.right  = w.rect.left
                if dx < 0: self.rect.left   = w.rect.right
            self.rect.y += dy
            for w in pygame.sprite.spritecollide(self, walls, False):
                if dy > 0: self.rect.bottom = w.rect.top
                if dy < 0: self.rect.top    = w.rect.bottom

    class Exit(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface((TILE_SIZE, TILE_SIZE))
            self.image.fill("dodgerblue")
            self.rect = self.image.get_rect(topleft=(x, y))

    class Back(pygame.sprite.Sprite):
        def __init__(self, x, y):
            super().__init__()
            self.image = pygame.Surface((TILE_SIZE, TILE_SIZE))
            self.image.fill("orange")
            self.rect = self.image.get_rect(topleft=(x, y))

    # ---- Anzahl Level (für Modulo) ----
    ANZAHL_LEVELS = 2

    # ---- Level-eigene Methoden ----

    def __init__(self, level_nr):
        self.walls   = pygame.sprite.Group()
        self.coins   = pygame.sprite.Group()
        self.enemies = pygame.sprite.Group()
        self.exits   = pygame.sprite.Group()
        self.backs   = pygame.sprite.Group()
        self.player  = None

        # Passende Aufbau-Methode aufrufen
        if level_nr == 0:
            self._level_0()
        elif level_nr == 1:
            self._level_1()

    def _level_0(self):
        T = TILE_SIZE
        # Rahmen: obere und untere Wand
        for x in range(0, 10 * T, T):
            self.walls.add(Level.Wall(x, 0))
            self.walls.add(Level.Wall(x, 6 * T))
        # Rahmen: linke und rechte Wand
        for y in range(T, 6 * T, T):
            self.walls.add(Level.Wall(0, y))
            self.walls.add(Level.Wall(9 * T, y))
        # Innenwand
        for x in range(3 * T, 7 * T, T):
            self.walls.add(Level.Wall(x, 3 * T))

        self.player = Level.Player(2 * T, 1 * T)
        self.coins.add(Level.Coin(7 * T, 2 * T))
        self.coins.add(Level.Coin(5 * T, 5 * T))
        self.enemies.add(Level.Enemy(4 * T, 5 * T))
        self.exits.add(Level.Exit(8 * T, 5 * T))

    def _level_1(self):
        T = TILE_SIZE
        # Größeres Level, ähnliches Muster ...
        for x in range(0, 14 * T, T):
            self.walls.add(Level.Wall(x, 0))
            self.walls.add(Level.Wall(x, 8 * T))
        for y in range(T, 8 * T, T):
            self.walls.add(Level.Wall(0, y))
            self.walls.add(Level.Wall(13 * T, y))

        self.player = Level.Player(2 * T, 2 * T)
        self.coins.add(Level.Coin(6 * T, 3 * T))
        self.coins.add(Level.Coin(10 * T, 5 * T))
        self.enemies.add(Level.Enemy(8 * T, 2 * T))
        # Breiter Ausgang (2 Tiles)
        self.exits.add(Level.Exit(12 * T, 6 * T))
        self.exits.add(Level.Exit(12 * T, 7 * T))
        # Rückgang (2 Tiles am linken Rand)
        self.backs.add(Level.Back(1 * T, 6 * T))
        self.backs.add(Level.Back(1 * T, 7 * T))

    def update(self):
        self.player.update(self.walls)
        self.enemies.update(self.walls)
        gesammelt = pygame.sprite.spritecollide(
            self.player, self.coins, True
        )
        return len(gesammelt)

    def draw(self, screen):
        self.walls.draw(screen)
        self.coins.draw(screen)
        self.enemies.draw(screen)
        self.exits.draw(screen)
        self.backs.draw(screen)
        screen.blit(self.player.image, self.player.rect)

    def ausgang_erreicht(self):
        return pygame.sprite.spritecollideany(
            self.player, self.exits
        )

    def rueckgang_erreicht(self):
        return pygame.sprite.spritecollideany(
            self.player, self.backs
        )

Achtung: Verschachtelte Klassen sind in Python eher selten. In Java wären sie üblicher. Nutze sie nur, wenn die Sprite-Klassen wirklich nur im Level vorkommen.
Vorteil: Du kannst Objekte pixelgenau platzieren – sie müssen nicht im Raster liegen. Nachteil: Jedes Level ist eine eigene Methode mit vielen Koordinaten – wird bei großen Levels unübersichtlich.

Idee

Die einfachste Variante: Sprite-Klassen existieren wie gehabt global. Die Level-Klasse hat pro Level eine eigene Aufbau-Methode, die Objekte direkt per Koordinaten erzeugt. Keine Datei, kein Zeichenraster, keine verschachtelten Klassen.

Wann sinnvoll? Für schnelle Prototypen oder wenn du wenige, kleine Level hast und einfach loslegen willst.
